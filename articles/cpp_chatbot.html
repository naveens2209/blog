/*
reply is method of bot class which checks for the perfect output to query.
If match is found, output stored below initial input is used.
If the output cannot be found, bot repeats input and prompts User to enter an ideal output.
This output will be stored in a memory textfile using fstream.h headerfile along with initial input.
*/

class bot{
	string id, re, ideal, Name, UserName;
 public:
	//Constuctor to make object of bot.
  bot (string U){
	  Name == "V";
		UserName == U;
	}
	void reply(string input)   // Teach AI to reply.
	{
	  fstream M;
	  M.open("memory.txt", ios::in);    // Open the textfile for input.
    // Search through the file until end is reached.
    while( !M.eof() ) {
      getline(M,id);    // Get next input.
		  // Check whether id is what we are looking for.
		  if(id == input) {
	      getline(M,re);   // If so, get related reply.
	      cout << re ;
	      return;
	    }
    }
    M.close();    // If reply couldn't find the input in the textfile. Close it!
    M.open("memory.txt", ios::out | ios::app);    // Open memory.txt for output, and use append at end of file.
    M << input << endl;   // Record initial input in memory.
    cout << "I don't know what you mean when you say \'" << input << "\', what should I say?\n\tIDEAL REPLY (PLEASE INPUT) : ";
    getline(cin,ideal);    // Get the ideal output.
    M << ideal << endl;    // Write the ideal output to memory.
    M.close();    // Close the file!
  }

  void whatname() {
    cout << Name;
  }
  void whatun(){
    cout << UserName;
  }
};
